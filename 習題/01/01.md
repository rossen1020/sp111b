#include <stdio.h>

double integrate(double (*f)(double), double x, double y) {<br>
    double sum=0.0;<br>
    for(double i = x;i<y;i+=0.0001){<br>
        sum+=f(i)*0.0001;<br>
    }<br>
    return sum;<br>
}

double square(double x) {<br>
    return x*x;<br>
}

int main() {<br>
    printf("integrate(square, 0.0, 2.0)=%f\n", integrate(square, 0.0, 2.0));<br>
}<br>
